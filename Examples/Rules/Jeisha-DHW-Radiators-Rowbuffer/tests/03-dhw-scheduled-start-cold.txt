# Scheduled DHW start at 13:00; cold weather (<5C) prepends a forced
# defrost and delays the mode switch by 540s.
time 12 59
silent Operating_Mode_State 0
silent Heatpump_State 1
silent Max_Pump_Duty 112
silent DHW_Temp 30
silent DHW_Target_Temp 48
silent DHW_Heat_Delta -8
silent Room_Thermostat_Temp 21
silent Outside_Temp 2
boot
expectcmd SetMaxPumpDuty 112
expectcmd SetPump 0
adv 75
expectcmd SetHeatpump 1
expectcmd SetForceDefrost 1
expectnone
