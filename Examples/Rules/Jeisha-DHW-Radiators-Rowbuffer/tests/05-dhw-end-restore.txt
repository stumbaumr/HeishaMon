# Valve back to heating: staged restore of heat request/curves (240s),
# pump duty (270s), and operation mode/heatpump state (300s).
time 13 5
silent Heatpump_State 1
silent ThreeWay_Valve_State 1
silent Operating_Mode_State 0
silent Max_Pump_Duty 112
silent DHW_Temp 45
silent DHW_Target_Temp 48
silent DHW_Heat_Delta -8
silent Room_Thermostat_Temp 21
silent Outside_Temp 10
boot
expectcmd SetMaxPumpDuty 112
expectcmd SetPump 0
set ThreeWay_Valve_State 0
adv 240
expectcmd SetZ1HeatRequestTemperature 5
expectcmd SetCurves {zone1:{heat:{target:{high:40,low:31}}}}
adv 30
expectcmd SetMaxPumpDuty 112
expectcmd SetPump 0
adv 30
expectcmd SetOperationMode 0
expectcmd SetHeatpump 1
expectnone
